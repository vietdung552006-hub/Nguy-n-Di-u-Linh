<?php
include '../includes/auth_check.php';
include '../config/db.php';

$user_id = $_SESSION['user_id'];

$status_filter = isset($_GET['status']) ? $_GET['status'] : '';
$sort = isset($_GET['sort']) && $_GET['sort'] == 'due_date' ? 'due_date ASC' : 'created_at DESC';

$query = "SELECT * FROM tasks WHERE user_id = :user_id";
if ($status_filter) {
    $query .= " AND status = :status";
}
$query .= " ORDER BY $sort";

$stmt = $pdo->prepare($query);
$params = ['user_id' => $user_id];
if ($status_filter) $params['status'] = $status_filter;
$stmt->execute($params);
$tasks = $stmt->fetchAll(PDO::FETCH_ASSOC);

if (isset($_GET['success'])) {
    $alert = '<div class="alert alert-success">Thành công: ' . ($_GET['success'] == 'added' ? 'Thêm' : ($_GET['success'] == 'updated' ? 'Cập nhật' : 'Xóa')) . ' công việc!</div>';
} elseif (isset($_GET['error'])) {
    $alert = '<div class="alert alert-danger">Lỗi: Công việc không tồn tại hoặc lỗi hệ thống.</div>';
}
?>

<?php include '../includes/header.php'; ?>
<h2>Danh sách công việc</h2>
<?php if (isset($alert)) echo $alert; ?>
<a href="add_task.php" class="btn btn-success mb-3">Thêm công việc mới</a>

<form method="GET" class="mb-3">
    <div class="row">
        <div class="col-md-4">
            <label class="form-label">Lọc theo trạng thái</label>
            <select name="status" class="form-select">
                <option value="">Tất cả</option>
                <option value="pending" <?php if ($status_filter == 'pending') echo 'selected'; ?>>Pending</option>
                <option value="in_progress" <?php if ($status_filter == 'in_progress') echo 'selected'; ?>>In Progress</option>
                <option value="completed" <?php if ($status_filter == 'completed') echo 'selected'; ?>>Completed</option>
            </select>
        </div>
        <div class="col-md-4">
            <label class="form-label">Sắp xếp</label>
            <select name="sort" class="form-select">
                <option value="">Mới nhất</option>
                <option value="due_date" <?php if (isset($_GET['sort']) && $_GET['sort'] == 'due_date') echo 'selected'; ?>>Theo ngày hết hạn</option>
            </select>
        </div>
        <div class="col-md-4 d-flex align-items-end">
            <button type="submit" class="btn btn-primary">Áp dụng</button>
        </div>
    </div>
</form>

<table class="table table-striped">
    <thead>
        <tr>
            <th>Tiêu đề</th>
            <th>Mô tả</th>
            <th>Ngày hết hạn</th>
            <th>Trạng thái</th>
            <th>Ngày tạo</th>
            <th>Hành động</th>
        </tr>
    </thead>
    <tbody>
        <?php if (count($tasks) > 0): ?>
            <?php foreach ($tasks as $row): ?>
                <tr>
                    <td><?php echo htmlspecialchars($row['title']); ?></td>
                    <td><?php echo htmlspecialchars($row['description'] ?? ''); ?></td>
                    <td><?php echo $row['due_date'] ? date('d/m/Y', strtotime($row['due_date'])) : ''; ?></td>
                    <td><span class="badge <?php echo $row['status'] == 'completed' ? 'bg-success' : ($row['status'] == 'in_progress' ? 'bg-info' : 'bg-warning'); ?>"><?php echo ucfirst($row['status']); ?></span></td>
                    <td><?php echo date('d/m/Y H:i', strtotime($row['created_at'])); ?></td>
                    <td>
                        <a href="edit_task.php?id=<?php echo $row['id']; ?>" class="btn btn-warning btn-sm">Sửa</a>
                        <a href="delete_task.php?id=<?php echo $row['id']; ?>" class="btn btn-danger btn-sm" onclick="return confirm('Xác nhận xóa?')">Xóa</a>
                    </td>
                </tr>
            <?php endforeach; ?>
        <?php else: ?>
            <tr><td colspan="6" class="text-center">Chưa có công việc nào. Hãy thêm mới!</td></tr>
        <?php endif; ?>
    </tbody>
</table>
<?php include '../includes/footer.php'; ?>
