<?php
include '../includes/auth_check.php';
include '../config/db.php';

if (isset($_GET['id'])) {
    $id = (int)$_GET['id'];
    $user_id = $_SESSION['user_id'];

    $sql = "SELECT * FROM tasks WHERE id = $id AND user_id = $user_id";
    $result = $conn->query($sql);
    if ($result->num_rows == 0) {
        header("Location: index.php?error=not_found");
        exit();
    }
    $task = $result->fetch_assoc();
}

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $title = mysqli_real_escape_string($conn, $_POST['title']);
    $description = mysqli_real_escape_string($conn, $_POST['description']);
    $status = $_POST['status'];

    $sql = "UPDATE tasks SET title = '$title', description = '$description', status = '$status' WHERE id = $id AND user_id = $user_id";
    if ($conn->query($sql) === TRUE) {
        header("Location: index.php?success=updated");
        exit();
    } else {
        $error = "Lỗi: " . $conn->error;
    }
}
?>

<?php include '../includes/header.php'; ?>
<h2>Sửa công việc</h2>
<?php if (isset($error)): ?><div class="alert alert-danger"><?php echo $error; ?></div><?php endif; ?>
<form method="POST">
    <div class="mb-3">
        <label class="form-label">Tiêu đề</label>
        <input type="text" name="title" class="form-control" value="<?php echo htmlspecialchars($task['title']); ?>" required>
    </div>
    <div class="mb-3">
        <label class="form-label">Mô tả</label>
        <textarea name="description" class="form-control" rows="4" required><?php echo htmlspecialchars($task['description']); ?></textarea>
    </div>
    <div class="mb-3">
        <label class="form-label">Trạng thái</label>
        <select name="status" class="form-select">
            <option value="pending" <?php if ($task['status'] == 'pending') echo 'selected'; ?>>Chưa hoàn thành</option>
            <option value="completed" <?php if ($task['status'] == 'completed') echo 'selected'; ?>>Hoàn thành</option>
        </select>
    </div>
    <button type="submit" class="btn btn-primary">Cập nhật</button>
</form>
<?php include '../includes/footer.php'; ?>
